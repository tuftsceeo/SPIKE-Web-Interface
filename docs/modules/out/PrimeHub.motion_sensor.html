     if ( newGesture == 'tapped') {
            console.log("SPIKE was tapped")
     }
     else if ( newGesture == 'doubletapped') {
            console.log("SPIKE was doubletapped")
     }
     else if ( newGesture == 'shaken') {
            console.log("SPIKE was shaken")

     else if ( newGesture == 'freefall') {
